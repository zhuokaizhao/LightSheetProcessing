334 inline std::vector<std::string> split_up(std::string str) {
335 
336  std::vector<char> delims = {'\'', '\"'};
337  auto find_ws = [](char ch) { return std::isspace<char>(ch, std::locale()); };
338  trim(str);
339 
340  std::vector<std::string> output;
341 
342  while(!str.empty()) {
343  if(str[0] == '\'') {
344  auto end = str.find('\'', 1);
345  if(end != std::string::npos) {
346  output.push_back(str.substr(1, end - 1));
347  str = str.substr(end + 1);
348  } else {
349  output.push_back(str.substr(1));
350  str = "";
351  }
352  } else if(str[0] == '\"') {
353  auto end = str.find('\"', 1);
354  if(end != std::string::npos) {
355  output.push_back(str.substr(1, end - 1));
356  str = str.substr(end + 1);
357  } else {
358  output.push_back(str.substr(1));
359  str = "";
360  }
361 
362  } else {
363  auto it = std::find_if(std::begin(str), std::end(str), find_ws);
364  if(it != std::end(str)) {
365  std::string value = std::string(str.begin(), it);
366  output.push_back(value);
367  str = std::string(it, str.end());
368  } else {
369  output.push_back(str);
370  str = "";
371  }
372  }
373  trim(str);
374  }
375 
376  return output;
377 }
